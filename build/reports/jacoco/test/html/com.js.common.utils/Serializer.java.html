package com.js.common.utils;

/**
 * @author goutam.mandal
 *
 */
public interface Serializer<O> {
	
	public O serialize(Object object);

	public <I> I deserialize(O object, Class<? extends I> clazz);

	public enum FieldNameFormat {
		/**
		 * e.g., "lowerCamel".
		 */
		CAMEL_CASE,

		/**
		 * e.g., "UPPER_UNDERSCORE".
		 */
		UPPERCASE_UNDERSCORE, ;
	}

	public enum NullPolicy {
		/**
		 * Value = null
		 */
		NULL,

		/**
		 * Value = "null"
		 */
		@Deprecated
		NULL_STRING,

		/**
		 * Value = ""
		 */
		@Deprecated
		EMPTY_STRING,

		/**
		 * Field will not appear in resulting Json
		 */
		OMIT_FIELD, ;
	}

}
