package com.js.common.utils;

import java.nio.charset.Charset;
import java.nio.charset.StandardCharsets;

/**
 * @author goutam.mandal
 */
public class CharEncodingUtil {

	private CharEncodingUtil() {
		throw new UnsupportedOperationException("Utility Class !");
	}

	/**
	 * @param string
	 * @param fromEncoding
	 * @param toEncoding
	 * @return
	 */
	public static String convert(String string, Charset fromEncoding, Charset toEncoding) {
		if (string == null) {
			return null;
		}
		return new String(string.getBytes(fromEncoding), toEncoding);
	}

	/**
	 * @param string
	 * @param charsets Possible charsets
	 * @return
	 */
	public static Charset detectCharset(String string, Charset charsets[]) {
		Charset probe = StandardCharsets.UTF_8;
		if (string == null) {
			return probe;
		}
		for (Charset charset : charsets) {
			if (charset != null && string.equals(convert(convert(string, charset, probe), probe, charset))) {
					return charset;
			}
		}
		return probe;
	}

	private static final Charset POSSIBLE_CHARSETS[] = new Charset[] { StandardCharsets.ISO_8859_1, StandardCharsets.UTF_8 };

	/**
	 * Will be tested against {@link CharEncodingUtil#POSSIBLE_CHARSETS}
	 * @param string
	 * @return
	 */
	public static Charset detectCharset(String string) {
		return detectCharset(string, POSSIBLE_CHARSETS);
	}
}
